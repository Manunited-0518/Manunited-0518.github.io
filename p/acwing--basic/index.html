基础算法

快速排序

（基于分治的思想）

算法思想

  • 第一步：确定分界点。随机找一个元素作为分界点（一般找第一个,也可以是中间值或者随机）
  • 第二步：调整区间。让第一个区间都小于等于中间值，让第二个区间都大于等于中间值。
  • 第三步：递归处理左右两段。

第二步很关键，所以如何处理第二步是最大的问题。

这里给出一个最易于理解的方法，只是需要额外的空间，可以在忘记其他做法的时候使用此方法。

  • 第一步：建立两个数组，a[ ]和b[ ]
  • 第二步：扫描q[ l - r ]，把小于中间值的放到a，其余放到b
  • 第三步：把a放到q，再把b放到q

较为优美的方法，即为王道讲的方法。

#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n;
int q[N];

void quick_sort(int q[],int l,int r){
  if(l >=r) return;
  
  int x =q[l],i=l-1,j=r+1;
  //为什么不是i=l，j=r？
  //和之后的写法有关，此写法会将两个指针开始就向中间移动一次，所以要把指针放到数组的“两侧”
  while(i < j){
    do i++ ; while(q[i] < x);
    do j-- ; while(q[j] > x);
    if(i < j) swap(q[i],q[j]);
  }
  quick_sort(q,l,j);
  quick_sort(q,j + 1,r);
}

int main()
{
  scanf("%d",&n);
  for(int i+0; i<n ;i++) scanf("%d",&q[i]);
  
  quick_sort(q, 0, n-1);
  
  for (int i=0; i < n; i ++) printf("%d",q[i]);
  
  return 0;
}

归并排序

（基于分治的思想）

时间复杂度：nlog2n

算法思想

  • 第一步：找分界点。mid=（1+r）/2
  • 第二步：递归排序left和right
  • 第三步：归并。合二为一

最难处理的为第三步，下面是第三步的处理方法

  • 第一步：建立两个指针，分别指向两个序列的最小值
  • 第二步：比较两个指针的值，将较小者输出，并指向下一个
  • 第三步：循环执行第二步，直至一个序列的值输出完，将另一个序列的值全部输出

void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);

    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];

    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];

    for (i = l, j = 0; i <= r; i ++, j ++ ) q[i] = tmp[j];
}

二分排序

整数二分

有单调性一定可以二分，但二分不一定有单调性

本质：找到一个性质，可以把区间分为两半

性质：整数二分一定有解，根据题目去判断无解的情况

算法思想

  1. 如果check是不是在左半边：

  • 求mid=(l+r+1)/2

    If( check( mid ) )

    • 若check为True: 区间为[mid,r]，更新方式为l=mid
    • 若check为false: 区间为[l,mid-1]，更新方式为r=mid-1
  1. 如果check是不是在右半边：

  • 求mid=(l+r)/2

    If( check( mid ) )

    • 若check为True: 区间为[l,mid]，更新方式为r=mid
    • 若check为false: 区间为[mid+1,r]，更新方式为l=mid+1

bool check(int x) {/* ... */} // 检查x是否满足某种性质

// 区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用：
int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;    // check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
// 区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用：
int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

double bsearch_3(double l, double r)
{
    const double eps = 1e-6;   // eps 表示精度，取决于题目对精度的要求
    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        if (check(mid)) r = mid;
        else l = mid;
    }
    return l;
}

    C++11中添加了一种新特性——范围遍历

    C++11 引入了一种新的方式来遍历容器，称为 范围基 for 循环（range-based for loop）。它使得遍历容器变得更加简洁和易读。

    语法：

    1

3

    // 代码块

    ​ • declaration 是你想要访问容器元素的声明。可以是值，也可以是引用。

    ​ • container 是你要遍历的容器，比如 vector、array、list 等。

    特点：

    ​ 1. 简洁：范围基 for 循环自动处理了迭代器或下标，避免了传统 for 循环中手动增加迭代器或下标的繁琐。

    ​ 2. 安全：对于容器元素，范围基 for 循环会确保访问的每个元素都是有效的，并且不会发生越界。

    ​ 3. 支持任何容器：不仅限于数组，vector、list、set 等容器都可以使用范围基 for 循环。

        if (i < B.size()) t += B[i];

}

    位运算

    • 基本问题：n的二进制表示中第k位是几

    • 基本思路：

      • 先把第k位移到最后一位—n » k
      • 看个位是几—x & 1

    • Lowbit：

      • lowbit(x)：返回x的最后一位1（以二进制的形式返回）

      • ～x：x取反

      • -x=～x+1

      • x & -x：这就是lowbit的执行过程

      • 应用：

        • 计算出x中1的个数

    例题（二进制中1的个数）

    给定一个长度为n的数列，请你求出数列中每个数的二进制表示中1的个数。

    输入格式

    第一行包含整数n。

    第二行包含n个整数，表示整个数列。

    输出格式

    共一行，包含n个整数，其中的第i个数表示数列中的第i个数的二进制表示中1的个数。

    **

    数据范围

typedef pair<int, int> PII;

int n;
vector<PII> segs;

void merge(vector<PII> &segs)
{
  vector<PII> res;
  
  sort(segs.begin(), segs.end());
  
  int st = -2e9, ed = -2e9;
  for (auto seg : segs)
    if (ed < seg.first)
    {
      if (st != -2e9) res.push_back({st, ed});
      st = seg.first, ed = seg.second;
    }
    else ed = max(ed, seg.second);
  
  if (st != -2e9) res.push_back({st, ed});
  
  segs = res;
}

  for (int i = 0; i < n; i ++ )
  {
    int l,r;
    cin >> l >> r;
    segs.push_back({l,r});
  }
  
  merge(segs);
  
  cout << segs.size() << endl;
  
  return 0;
}

const int N = 100010;

int n, k;
int q[N];

int quick_sort(int l, int r, int k)
{
  if (l >= r) return q[l];
  
  int x = q[l], i = l - 1, j = r + 1;
  while (i < j)
  {
    while (q[ ++ i ] < x);
    while (q[ -- j ] > x);
    if (i < j) swap(q[i], q[j]);
  }
  
  int sl = j - l + 1;
  if (k <= sl) return quick_sort(l, j, k);
  
  return quick_sort(j + 1, r, k - sl);
}

    if (i >= k - 1) printf("%d ", a[q[hh]]);
  }
  puts("");
  
  hh = 0, tt = -1;
  for ( int i = 0; i < n; i ++ )
  {
    //判断队头是否已经滑出窗口
    if (hh <= tt && i - k + 1 > q[hh]) h ++ ;
    while (hh <= tt && a[q[tt]] >= a[i]) tt -- ;
    q[ ++ tt] = i;
    if (i >= k - 1) printf("%d ", a[q[hh]]);
  }
  puts("");
  
  return 0;
}

数据范围 1 ≤ N ≤ 104

1 ≤ M ≤105

using namespace std;

const int N = 10010, M = 100010;

int n, m;
char p[N], s[M];
int ne[N];

int main()
{
  cin >> n >> p + 1 >> m >> s + 1;
  
  // 求next的过程
  for (int i = 2, j = 0; i <= n; i ++ )
  {
    while (j && p[i] != p[j + 1]) j = ne[j];
    if (p[i] == p[j + 1]) j ++ ;
    ne[i] = j;
  }
  
  // KMP匹配过程
  for (int i = 1, j = 0; i <= m; i ++ )
  {
    while (j && s[i] != p[j + 1]) j = ne[j];
    if (s[i] == p[j + 1]) j ++ ;
    if (j == n)
    {
      printf("%d ", i - n);
      j = ne[j]; 
    }
  }
}
